# Chronos Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Rust CLI that provides a human-friendly interface over system `crontab` and `at`, with natural language schedule parsing, table/JSON output, and a lightweight metadata sidecar.

**Architecture:** No job store — system crontab and at queues are source of truth. A metadata sidecar TOML file at `~/.config/chronos/meta.toml` stores optional id/description/source for jobs added through Chronos. Schedule input supports both cron expressions and natural language, with auto-detection of recurring vs one-off.

**Tech Stack:** Rust, clap (CLI), chrono + chrono-english (date/time), croner + cron-descriptor (cron parsing/humanizing), comfy-table (tables), toml + toml_edit (sidecar), serde + serde_json (serialization), nucleo-matcher (fuzzy search)

```rust
use anyhow::Result;
use std::process::Command;

/// A parsed crontab entry.
#[derive(Debug, Clone)]
pub struct CrontabEntry {
    pub schedule: String,
    pub command: String,
    pub enabled: bool,
    /// The full original line (for matching in sidecar).
    pub raw_line: String,
}

/// Parse crontab text into entries. Skips pure comment lines (not disabled jobs)
/// and environment variable assignments.
pub fn parse_crontab(content: &str) -> Vec<CrontabEntry> {
    let mut entries = Vec::new();

    for line in content.lines() {
        let trimmed = line.trim();
        if trimmed.is_empty() {
            continue;
        }

        // Check for disabled cron entry: starts with # followed by a cron schedule
        if let Some(stripped) = trimmed.strip_prefix('#') {
            let stripped = stripped.trim_start();
            if let Some(entry) = try_parse_cron_line(stripped) {
                entries.push(CrontabEntry {
                    schedule: entry.0,
                    command: entry.1,
                    enabled: false,
                    raw_line: format!("{} {}", entry.0, entry.1),
                });
                continue;
            }
            // Pure comment, skip
            continue;
        }

        // Skip environment variable lines (KEY=value)
        if trimmed.contains('=') && !trimmed.starts_with(|c: char| c.is_ascii_digit() || c == '*') {
            continue;
        }

        if let Some(entry) = try_parse_cron_line(trimmed) {
            entries.push(CrontabEntry {
                schedule: entry.0.clone(),
                command: entry.1.clone(),
                enabled: true,
                raw_line: format!("{} {}", entry.0, entry.1),
            });
        }
    }

    entries
}

/// Try to parse a line as a 5-field cron schedule + command.
/// Returns (schedule, command) if successful.
fn try_parse_cron_line(line: &str) -> Option<(String, String)> {
    let parts: Vec<&str> = line.splitn(6, char::is_whitespace).collect();
    if parts.len() < 6 {
        return None;
    }
    // Validate that the first 5 parts look like cron fields
    let schedule_parts = &parts[..5];
    for part in schedule_parts {
        if part.is_empty() {
            return None;
        }
    }
    let schedule = schedule_parts.join(" ");
    let command = parts[5..].join(" ").trim().to_string();
    if command.is_empty() {
        return None;
    }
    Some((schedule, command))
}

/// Add a new entry to crontab content.
pub fn add_crontab_entry(content: &str, schedule: &str, command: &str) -> String {
    let mut result = content.trim_end().to_string();
    if !result.is_empty() {
        result.push('\n');
    }
    result.push_str(&format!("{schedule} {command}\n"));
    result
}

/// Remove an entry matching the given raw line from crontab content.
pub fn remove_crontab_entry(content: &str, raw_line: &str) -> String {
    content
        .lines()
        .filter(|line| {
            let trimmed = line.trim();
            // Don't match the exact line or its disabled form
            trimmed != raw_line && trimmed != format!("#{raw_line}")
        })
        .collect::<Vec<_>>()
        .join("\n")
        + "\n"
}

/// Disable an entry by commenting it out.
pub fn disable_crontab_entry(content: &str, raw_line: &str) -> String {
    content
        .lines()
        .map(|line| {
            if line.trim() == raw_line {
                format!("#{line}")
            } else {
                line.to_string()
            }
        })
        .collect::<Vec<_>>()
        .join("\n")
        + "\n"
}

/// Enable an entry by uncommenting it.
pub fn enable_crontab_entry(content: &str, raw_line: &str) -> String {
    content
        .lines()
        .map(|line| {
            let trimmed = line.trim();
            if trimmed.strip_prefix('#').map(|s| s.trim_start()) == Some(raw_line) {
                raw_line.to_string()
            } else {
                line.to_string()
            }
        })
        .collect::<Vec<_>>()
        .join("\n")
        + "\n"
}

/// Read the current user's crontab. Returns empty string if no crontab set.
pub fn read_system_crontab() -> Result<String> {
    let output = Command::new("crontab").arg("-l").output()?;
    if output.status.success() {
        Ok(String::from_utf8_lossy(&output.stdout).to_string())
    } else {
        // "no crontab for user" is not an error for us
        Ok(String::new())
    }
}

/// Write content to the current user's crontab.
pub fn write_system_crontab(content: &str) -> Result<()> {
    use std::io::Write;
    let mut child = Command::new("crontab")
        .arg("-")
        .stdin(std::process::Stdio::piped())
        .spawn()?;
    child.stdin.as_mut().unwrap().write_all(content.as_bytes())?;
    let status = child.wait()?;
    if !status.success() {
        anyhow::bail!("crontab write failed with status {status}");
    }
    Ok(())
}
```

```rust
use anyhow::Result;
use std::process::Command;

#[derive(Debug, Clone)]
pub struct AtEntry {
    pub job_number: u32,
    pub scheduled_time: String,
    pub command: Option<String>,
}

/// Parse output from `atq` into entries.
pub fn parse_atq_output(output: &str) -> Vec<AtEntry> {
    output
        .lines()
        .filter_map(|line| {
            let line = line.trim();
            if line.is_empty() {
                return None;
            }
            let mut parts = line.splitn(2, '\t');
            let job_number: u32 = parts.next()?.parse().ok()?;
            let rest = parts.next().unwrap_or("");
            // Rest is like "Mon Mar 30 15:00:00 2026 a droshima"
            // Extract the datetime portion (everything before the queue letter + user)
            let scheduled_time = rest
                .rsplitn(3, ' ')
                .collect::<Vec<_>>()
                .into_iter()
                .rev()
                .next()
                .unwrap_or(rest)
                .to_string();
            Some(AtEntry {
                job_number,
                scheduled_time: rest.to_string(),
                command: None,
            })
        })
        .collect()
}

/// Check if `at` is available by parsing a success/fail flag.
pub fn parse_at_availability_check(success: bool) -> bool {
    success
}

/// Check if `at` command is available on this system.
pub fn is_at_available() -> bool {
    Command::new("which")
        .arg("at")
        .output()
        .map(|o| o.status.success())
        .unwrap_or(false)
}

/// Extract the user's command from `at -c` output.
/// The script contains shell setup (exports, etc.) — we want the last non-empty line(s)
/// that aren't shell boilerplate.
pub fn extract_command_from_at_script(script: &str) -> String {
    let lines: Vec<&str> = script.lines().collect();
    // Skip shebang, comments, and export lines; take the remaining content
    let command_lines: Vec<&str> = lines
        .iter()
        .filter(|line| {
            let trimmed = line.trim();
            !trimmed.is_empty()
                && !trimmed.starts_with("#!")
                && !trimmed.starts_with('#')
                && !trimmed.starts_with("export ")
        })
        .copied()
        .collect();
    command_lines.join("\n")
}

/// Read pending at jobs.
pub fn read_at_queue() -> Result<Vec<AtEntry>> {
    let output = Command::new("atq").output()?;
    if !output.status.success() {
        return Ok(Vec::new());
    }
    let text = String::from_utf8_lossy(&output.stdout);
    let mut entries = parse_atq_output(&text);

    // Fetch the command for each entry
    for entry in &mut entries {
        if let Ok(output) = Command::new("at").arg("-c").arg(entry.job_number.to_string()).output() {
            if output.status.success() {
                let script = String::from_utf8_lossy(&output.stdout);
                entry.command = Some(extract_command_from_at_script(&script));
            }
        }
    }

    Ok(entries)
}

/// Schedule a one-off job with `at`.
pub fn schedule_at_job(datetime: &str, command: &str) -> Result<u32> {
    use std::io::Write;
    let mut child = Command::new("at")
        .arg(datetime)
        .stdin(std::process::Stdio::piped())
        .stderr(std::process::Stdio::piped())
        .spawn()?;
    child.stdin.as_mut().unwrap().write_all(command.as_bytes())?;
    let output = child.wait_with_output()?;
    if !output.status.success() {
        let stderr = String::from_utf8_lossy(&output.stderr);
        anyhow::bail!("at failed: {stderr}");
    }
    // at prints "job N at <datetime>" to stderr
    let stderr = String::from_utf8_lossy(&output.stderr);
    let job_number = stderr
        .split_whitespace()
        .nth(1)
        .and_then(|s| s.parse::<u32>().ok())
        .unwrap_or(0);
    Ok(job_number)
}

/// Remove an at job.
pub fn remove_at_job(job_number: u32) -> Result<()> {
    let status = Command::new("atrm").arg(job_number.to_string()).status()?;
    if !status.success() {
        anyhow::bail!("atrm failed for job {job_number}");
    }
    Ok(())
}
```

```rust
    /// Load sidecar from file. Returns empty sidecar if file doesn't exist.
    pub fn load(path: &Path) -> Result<Self> {
        if !path.exists() {
            return Ok(Self::default());
        }
        let content = std::fs::read_to_string(path)?;
        let sidecar: Sidecar = toml::from_str(&content)?;
        Ok(sidecar)
    }

    /// Save sidecar to file, creating parent directories if needed.
    pub fn save(&self, path: &Path) -> Result<()> {
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)?;
        }
        let content = toml::to_string_pretty(self)?;
        std::fs::write(path, content)?;
        Ok(())
    }

    pub fn set_cron_meta(&mut self, raw_line: &str, meta: JobMeta) {
        self.cron.insert(raw_line.to_string(), meta);
    }

    pub fn get_cron_meta(&self, raw_line: &str) -> Option<&JobMeta> {
        self.cron.get(raw_line)
    }

    pub fn set_at_meta(&mut self, job_number: u32, meta: JobMeta) {
        self.at.insert(job_number, meta);
    }

    pub fn get_at_meta(&self, job_number: u32) -> Option<&JobMeta> {
        self.at.get(&job_number)
    }

    /// Remove sidecar entries that no longer correspond to active jobs.
    pub fn prune(&mut self, active_cron_lines: &[String], active_at_jobs: &[u32]) {
        self.cron.retain(|key, _| active_cron_lines.contains(key));
        self.at.retain(|key, _| active_at_jobs.contains(key));
    }

    /// Find a cron entry's raw line by its sidecar id.
    pub fn find_cron_by_id(&self, id: &str) -> Option<String> {
        self.cron
            .iter()
            .find(|(_, meta)| meta.id.as_deref() == Some(id))
            .map(|(key, _)| key.clone())

        self.at
            .iter()
            .find(|(_, meta)| meta.id.as_deref() == Some(id))
            .map(|(key, _)| *key)
```

```rust
use anyhow::{bail, Result};

/// Day-of-week mapping.
fn day_to_cron(day: &str) -> Option<&'static str> {
    match day {
        "sunday" | "sun" => Some("0"),
        "monday" | "mon" => Some("1"),
        "tuesday" | "tue" => Some("2"),
        "wednesday" | "wed" => Some("3"),
        "thursday" | "thu" => Some("4"),
        "friday" | "fri" => Some("5"),
        "saturday" | "sat" => Some("6"),
        _ => None,
    }
}

/// Parse a time expression like "8am", "8:30pm", "midnight", "noon" into (hour, minute).
fn parse_time(input: &str) -> Option<(u32, u32)> {
    let input = input.trim().to_lowercase();

    if input == "midnight" {
        return Some((0, 0));
    }
    if input == "noon" {
        return Some((12, 0));
    }

    // Handle "8am", "8pm", "8:30am", "8:30pm", "13:00"
    let (time_part, is_pm) = if input.ends_with("pm") {
        (input.trim_end_matches("pm"), true)
    } else if input.ends_with("am") {
        (input.trim_end_matches("am"), false)
    } else {
        (input.as_str(), false)
    };

    let (hour, minute) = if time_part.contains(':') {
        let parts: Vec<&str> = time_part.split(':').collect();
        let h: u32 = parts[0].parse().ok()?;
        let m: u32 = parts[1].parse().ok()?;
        (h, m)
    } else {
        let h: u32 = time_part.parse().ok()?;
        (h, 0)
    };

    let hour = if is_pm && hour < 12 {
        hour + 12
    } else if !is_pm && hour == 12 && input.ends_with("am") {
        0
    } else {
        hour
    };

    Some((hour, minute))
}

/// Convert a natural language recurring schedule to a cron expression.
pub fn natural_to_cron(input: &str) -> Result<String> {
    let input = input.trim().to_lowercase();

    // Handle bare keywords
    match input.as_str() {
        "hourly" => return Ok("0 * * * *".to_string()),
        "daily" => return Ok("0 0 * * *".to_string()),
        "weekly" => return Ok("0 0 * * 0".to_string()),
        "monthly" => return Ok("0 0 1 * *".to_string()),
        _ => {}
    }

    // Handle "every N minutes/hours"
    if input.starts_with("every ") {
        let rest = input.strip_prefix("every ").unwrap();

        // "every N minutes"
        if rest.ends_with("minutes") || rest.ends_with("minute") {
            let n: u32 = rest
                .split_whitespace()
                .next()
                .and_then(|s| s.parse().ok())
                .unwrap_or(0);
            if n > 0 && n <= 59 {
                return Ok(format!("*/{n} * * * *"));
            }
        }

        // "every N hours"
        if rest.ends_with("hours") || rest.ends_with("hour") {
            let n: u32 = rest
                .split_whitespace()
                .next()
                .and_then(|s| s.parse().ok())
                .unwrap_or(0);
            if n > 0 && n <= 23 {
                return Ok(format!("0 */{n} * * *"));
            }
        }

        // "every weekday at TIME"
        if rest.starts_with("weekday") {
            let time_str = rest
                .strip_prefix("weekday")
                .unwrap()
                .trim()
                .strip_prefix("at")
                .unwrap_or("")
                .trim();
            let (hour, minute) = parse_time(if time_str.is_empty() { "9am" } else { time_str })
                .unwrap_or((9, 0));
            return Ok(format!("{minute} {hour} * * 1-5"));
        }

        // "every day at TIME"
        if rest.starts_with("day ") {
            let time_str = rest
                .strip_prefix("day")
                .unwrap()
                .trim()
                .strip_prefix("at")
                .unwrap_or("")
                .trim();
            let (hour, minute) = parse_time(time_str).unwrap_or((0, 0));
            return Ok(format!("{minute} {hour} * * *"));
        }

        // "every sunday at TIME", "every monday at 8:30am"
        let words: Vec<&str> = rest.split_whitespace().collect();
        if let Some(cron_day) = words.first().and_then(|w| day_to_cron(w)) {
            let time_str = if words.len() > 2 && words[1] == "at" {
                words[2..].join(" ")
            } else if words.len() > 1 {
                words[1..].join(" ")
            } else {
                "0am".to_string()
            };
            let (hour, minute) = parse_time(&time_str).unwrap_or((0, 0));
            return Ok(format!("{minute} {hour} * * {cron_day}"));
        }
    }

    // Handle "daily at TIME"
    if input.starts_with("daily ") {
        let time_str = input
            .strip_prefix("daily")
            .unwrap()
            .trim()
            .strip_prefix("at")
            .unwrap_or("")
            .trim();
        let (hour, minute) = parse_time(time_str).unwrap_or((0, 0));
        return Ok(format!("{minute} {hour} * * *"));
    }

    bail!(
        "Could not parse recurring schedule: '{input}'. Try formats like 'every sunday at 6am', 'daily at 8am', 'every 2 hours'."
    )
}
```

```rust
/// Convert natural language one-off schedule to `at`-compatible time string.
fn natural_to_at_time(input: &str) -> Result<String> {
    let now = Local::now();
    let datetime = parse_date_string(input, now, Dialect::Us)
        .map_err(|e| anyhow::anyhow!("Could not parse one-off schedule '{input}': {e}"))?;
    // at accepts "HH:MM YYYY-MM-DD"
```

```rust
pub fn run(json: bool) -> Result<()> {
    let sidecar_path = sidecar::sidecar_path();
    let mut sidecar = Sidecar::load(&sidecar_path)?;
    let mut jobs = Vec::new();
    let mut index = 1;

    // Read crontab entries
    let crontab_content = crontab::read_system_crontab()?;
    let cron_entries = crontab::parse_crontab(&crontab_content);

    for entry in &cron_entries {
        let meta = sidecar.get_cron_meta(&entry.raw_line);
        let schedule_human = humanize_cron(&entry.schedule).unwrap_or_else(|_| entry.schedule.clone());
        jobs.push(Job {
            index,
            id: meta.and_then(|m| m.id.clone()),
            schedule_human,
            command: entry.command.clone(),
            source: meta
                .and_then(|m| m.source.clone())
                .unwrap_or_else(|| "unknown".to_string()),
            job_type: JobType::Cron,
            raw_schedule: entry.schedule.clone(),
            enabled: entry.enabled,
        });
        index += 1;
    }

    // Read at queue entries
    let at_entries = at::read_at_queue().unwrap_or_default();
    for entry in &at_entries {
        let meta = sidecar.get_at_meta(entry.job_number);
        jobs.push(Job {
            index,
            id: meta.and_then(|m| m.id.clone()),
            schedule_human: entry.scheduled_time.clone(),
            command: entry.command.clone().unwrap_or_else(|| "—".to_string()),
            source: meta
                .and_then(|m| m.source.clone())
                .unwrap_or_else(|| "unknown".to_string()),
            job_type: JobType::At,
            raw_schedule: String::new(),
            enabled: true,
        });
        index += 1;
    }

    // Prune stale sidecar entries
    let active_cron_lines: Vec<String> = cron_entries.iter().map(|e| e.raw_line.clone()).collect();
    let active_at_jobs: Vec<u32> = at_entries.iter().map(|e| e.job_number).collect();
    sidecar.prune(&active_cron_lines, &active_at_jobs);
    let _ = sidecar.save(&sidecar_path);

    // Output
    if json {
        println!("{}", output::format_json(&jobs)?);
    } else {
        println!("{}", output::format_table(&jobs));
    }

    Ok(())
}
```

```rust
use anyhow::Result;
use crate::schedule::{classify_schedule, ScheduleKind};
use crate::sidecar::{self, JobMeta, Sidecar};
use crate::system::{crontab, at};

pub fn run(
    schedule: String,
    command: String,
    id: Option<String>,
    desc: Option<String>,
    source: Option<String>,
    force: bool,
) -> Result<()> {
    let kind = classify_schedule(&schedule)?;
    let source = source.unwrap_or_else(|| {
        std::env::var("USER").unwrap_or_else(|_| "unknown".to_string())
    });

    match kind {
        ScheduleKind::Recurring { cron_expr } => {
            add_recurring(&cron_expr, &command, id, desc, &source, force)
        }
        ScheduleKind::OneOff { at_time } => {
            add_oneoff(&at_time, &command, id, desc, &source)
        }
    }
}

fn add_recurring(
    cron_expr: &str,
    command: &str,
    id: Option<String>,
    desc: Option<String>,
    source: &str,
    force: bool,
) -> Result<()> {
    let current = crontab::read_system_crontab()?;
    let raw_line = format!("{cron_expr} {command}");

    // Duplicate check
    if !force {
        let entries = crontab::parse_crontab(&current);
        if entries.iter().any(|e| e.raw_line == raw_line) {
            anyhow::bail!(
                "Duplicate job detected: '{raw_line}' already exists. Use --force to add anyway."
            );
        }
    }

    let new_content = crontab::add_crontab_entry(&current, cron_expr, command);
    crontab::write_system_crontab(&new_content)?;

    // Update sidecar
    if id.is_some() || desc.is_some() {
        let sidecar_path = sidecar::sidecar_path();
        let mut sidecar = Sidecar::load(&sidecar_path)?;
        sidecar.set_cron_meta(
            &raw_line,
            JobMeta {
                id: id.clone(),
                description: desc,
                source: Some(source.to_string()),
            },
        );
        sidecar.save(&sidecar_path)?;
    }

    let display_id = id.as_deref().unwrap_or(&raw_line);
    println!("Added recurring job: {display_id}");
    Ok(())
}

fn add_oneoff(
    at_time: &str,
    command: &str,
    id: Option<String>,
    desc: Option<String>,
    source: &str,
) -> Result<()> {
    if at::is_at_available() {
        let job_number = at::schedule_at_job(at_time, command)?;

        // Update sidecar
        let sidecar_path = sidecar::sidecar_path();
        let mut sidecar = Sidecar::load(&sidecar_path)?;
        sidecar.set_at_meta(
            job_number,
            JobMeta {
                id: id.clone(),
                description: desc,
                source: Some(source.to_string()),
            },
        );
        sidecar.save(&sidecar_path)?;

        let display_id = id.as_deref().unwrap_or(&job_number.to_string());
        println!("Added one-off job: {display_id} (at job #{job_number})");
    } else {
        add_oneoff_via_cron_fallback(at_time, command, id, desc, source)?;
    }
    Ok(())
}

fn add_oneoff_via_cron_fallback(
    at_time: &str,
    command: &str,
    id: Option<String>,
    desc: Option<String>,
    source: &str,
) -> Result<()> {
    // Parse at_time back to a datetime to build cron schedule
    let dt = chrono::NaiveDateTime::parse_from_str(at_time, "%H:%M %Y-%m-%d")
        .map_err(|e| anyhow::anyhow!("Failed to parse datetime '{at_time}': {e}"))?;

    let minute = dt.format("%M").to_string();
    let hour = dt.format("%H").to_string();
    let day = dt.format("%-d").to_string();
    let month = dt.format("%-m").to_string();

    let job_id = id.clone().unwrap_or_else(|| format!("oneoff-{}", dt.format("%s")));
    let cleanup_id = format!("{job_id}-cleanup");

    // Get path to chronos binary
    let chronos_bin = std::env::current_exe()?.to_string_lossy().to_string();

    // Job cron entry: wraps command with _run-once
    let job_cron = format!("{minute} {hour} {day} {month} *");
    let job_command = format!(
        "{chronos_bin} _run-once {job_id} --cleanup-id {cleanup_id} -- {command}"
    );

    // Cleanup cron entry: fires 1 minute later
    let cleanup_minute: u32 = minute.parse::<u32>().unwrap_or(0) + 1;
    let cleanup_cron = if cleanup_minute >= 60 {
        let cleanup_hour: u32 = hour.parse::<u32>().unwrap_or(0) + 1;
        format!("0 {cleanup_hour} {day} {month} *")
    } else {
        format!("{cleanup_minute} {hour} {day} {month} *")
    };
    let cleanup_command = format!(
        "{chronos_bin} remove {job_id} && {chronos_bin} remove {cleanup_id}"
    );

    // Write both entries
    let current = crontab::read_system_crontab()?;
    let with_job = crontab::add_crontab_entry(&current, &job_cron, &job_command);
    let with_cleanup = crontab::add_crontab_entry(&with_job, &cleanup_cron, &cleanup_command);
    crontab::write_system_crontab(&with_cleanup)?;

    // Sidecar for both entries
    let sidecar_path = sidecar::sidecar_path();
    let mut sidecar = Sidecar::load(&sidecar_path)?;
    sidecar.set_cron_meta(
        &format!("{job_cron} {job_command}"),
        JobMeta {
            id: Some(job_id.clone()),
            description: desc,
            source: Some(source.to_string()),
        },
    );
    sidecar.set_cron_meta(

            description: Some(format!("Cleanup for {job_id}")),
            source: Some("chronos".to_string()),
        },
    );
    sidecar.save(&sidecar_path)?;

    println!("Added one-off job: {job_id} (via self-destructing cron, at not available)");
    Ok(())
```

```rust
    let at_entries = at::read_at_queue().unwrap_or_default();

    let total_cron = cron_entries.len();

    if row_num == 0 {
        anyhow::bail!("Row numbers start at 1");
    }

    if row_num <= total_cron {
        // It's a cron entry
        let entry = &cron_entries[row_num - 1];
        let new_content = crontab::remove_crontab_entry(&crontab_content, &entry.raw_line);
        crontab::write_system_crontab(&new_content)?;
        sidecar.cron.remove(&entry.raw_line);
        sidecar.save(sidecar_path)?;
        println!("Removed cron job #{row_num}: {}", entry.command);
    } else if row_num <= total_cron + at_entries.len() {

        sidecar.at.remove(&entry.job_number);
        sidecar.save(sidecar_path)?;
```

```rust
use anyhow::Result;
use nucleo_matcher::pattern::{Atom, AtomKind, CaseMatching, Normalization};
use nucleo_matcher::{Config, Matcher, Utf32Str};
use crate::job::{Job, JobType};
use crate::output;
use crate::sidecar::{self, Sidecar};
use crate::system::{crontab, at};
use crate::schedule::humanize::humanize_cron;

pub fn run(query: &str, json: bool) -> Result<()> {
    let sidecar_path = sidecar::sidecar_path();
    let sidecar = Sidecar::load(&sidecar_path)?;

    // Build job list (same as list command)
    let mut jobs = Vec::new();
    let mut index = 1;

    let crontab_content = crontab::read_system_crontab()?;
    let cron_entries = crontab::parse_crontab(&crontab_content);

    for entry in &cron_entries {
        let meta = sidecar.get_cron_meta(&entry.raw_line);
        let schedule_human = humanize_cron(&entry.schedule).unwrap_or_else(|_| entry.schedule.clone());
        jobs.push(Job {
            index,
            id: meta.and_then(|m| m.id.clone()),
            schedule_human,
            command: entry.command.clone(),
            source: meta.and_then(|m| m.source.clone()).unwrap_or_else(|| "unknown".to_string()),
            job_type: JobType::Cron,
            raw_schedule: entry.schedule.clone(),
            enabled: entry.enabled,
        });
        index += 1;
    }

    let at_entries = at::read_at_queue().unwrap_or_default();
    for entry in &at_entries {
        let meta = sidecar.get_at_meta(entry.job_number);
        jobs.push(Job {
            index,
            id: meta.and_then(|m| m.id.clone()),
            schedule_human: entry.scheduled_time.clone(),
            command: entry.command.clone().unwrap_or_else(|| "—".to_string()),
            source: meta.and_then(|m| m.source.clone()).unwrap_or_else(|| "unknown".to_string()),
            job_type: JobType::At,
            raw_schedule: String::new(),
            enabled: true,
        });
        index += 1;
    }

    // Fuzzy match
    let mut matcher = Matcher::new(Config::DEFAULT);
    let pattern = Atom::new(query, CaseMatching::Ignore, Normalization::Smart, AtomKind::Fuzzy);

    let matched: Vec<Job> = jobs
        .into_iter()
        .filter(|job| {
            let searchable = format!(
                "{} {} {} {}",
                job.id.as_deref().unwrap_or(""),
                job.schedule_human,
                job.command,
                job.source,
            );
            let mut buf = Vec::new();
            let haystack = Utf32Str::new(&searchable, &mut buf);
            pattern.score(haystack, &mut matcher).is_some()
        })
        .collect();

    if json {
        println!("{}", output::format_json(&matched)?);
    } else {
        if matched.is_empty() {
            println!("No jobs matching '{query}'");
        } else {
            println!("{}", output::format_table(&matched));
        }
    }

    Ok(())
}
```

```rust
use anyhow::Result;
use std::process::Command;
use crate::commands::remove;

pub fn run(id: &str, cleanup_id: &str, command: Vec<String>) -> Result<()> {
    let full_command = command.join(" ");

    // Execute the original command
    let status = Command::new("sh")
        .arg("-c")
        .arg(&full_command)
        .status()?;

    // Regardless of command success, clean up both cron entries
    let _ = remove::run(id);
    let _ = remove::run(cleanup_id);

    if !status.success() {
        let code = status.code().unwrap_or(1);
        std::process::exit(code);
    }

    Ok(())
}
```
